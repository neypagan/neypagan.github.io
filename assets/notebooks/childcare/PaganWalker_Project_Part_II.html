# Define the categories for Ethnicity, Gender, and Age Group
ethnicities = ["Total Summary", "Black or African American", "Asian", "White", "Hispanic or Latino"]
genders = ["Unspecified", "Men", "Women"]
age_groups = [
    "16 years and over", "16 to 19 years", "16 to 17 years", "18 to 19 years",
    "20 years and over", "20 to 24 years", "25 years and over", "25 to 54 years", "55 years and over"
]

# Function to parse Ethnicity, Gender, and Age Group
def parse_and_fill_categories(value):
    ethnicity = None
    gender = None
    age_group = None

    # Check for Ethnicity
    for category in ethnicities:
        if category in value:
            ethnicity = category
            break

    # Check for Gender
    for subcat in genders:
        if subcat in value:
            gender = subcat
            break

    # Check for Age Group
    for age in age_groups:
        if age in value:
            age_group = age
            break

    return {
        "Ethnicity": ethnicity or "Uncategorized",
        "Gender": gender or "Unspecified",
        "Age Group": age_group or "Unspecified"
    }

# Apply the function to parse Ethnicity, Gender, and Age Group for all rows
parsed_categories = bls_pd_read["age_sex_race_ethnicity"].apply(parse_and_fill_categories).apply(pd.Series)
bls_pd_read = pd.concat([bls_pd_read, parsed_categories], axis=1)

# Function to correct Ethnicity
def correct_ethnicity(row_ethnicity, last_ethnicity):
    if isinstance(row_ethnicity, str) and row_ethnicity == "Uncategorized":
        return last_ethnicity
    return row_ethnicity

# Function to correct Gender
def correct_gender(row_gender, last_gender, row_ethnicity):
    if isinstance(row_gender, str) and row_gender == "Unspecified" and row_ethnicity != "Total Summary":
        return last_gender if last_gender else "Men"
    return row_gender

# Initialize previous values
last_ethnicity = None
last_gender = None

# Correct Ethnicity and Gender for all rows
corrected_ethnicities = []
corrected_genders = []

for _, row in bls_pd_read.iterrows():
    # Extract scalar values for row
    current_ethnicity = row["Ethnicity"]
    current_gender = row["Gender"]

    # Correct Ethnicity and Gender
    corrected_ethnicity = correct_ethnicity(current_ethnicity, last_ethnicity)
    corrected_gender = correct_gender(current_gender, last_gender, corrected_ethnicity)

    # Append the corrected values
    corrected_ethnicities.append(corrected_ethnicity)
    corrected_genders.append(corrected_gender)

    # Update last known values
    last_ethnicity = corrected_ethnicity
    last_gender = corrected_gender
